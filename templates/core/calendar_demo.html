Usage Examples
HTML Usage
<!-- Month Dropdown -->
<select class="form-select" id="card-exp-month">
    <option value="">Month</option>
    <option value="01">01</option>
    <!-- ... more options -->
</select>

<!-- Year Dropdown -->
<select class="form-select" id="card-exp-year">
    <option value="">Year</option>
    <option value="2024">2024</option>
    <!-- ... more options -->
</select>
JavaScript Usage
// Auto-initialization
// All selects with id containing 'exp-month' or 'month' 
// are automatically converted to month dropdowns

// Manual initialization
new CalendarDropdown(selectElement, {
    type: 'month', // or 'year'
    placeholder: 'Select Month'
});
